**现状问题**:

- 用户不知道当前位置

**优化方案**:

```vue
<!-- Breadcrumb.vue -->
<div class="breadcrumb">
  <router-link to="/" class="breadcrumb-item">
    <i class="icon-home"></i>首页
  </router-link>

  <router-link 
    v-for="(item, index) in breadcrumbs"

  return matched.map(r => ({
    path: r.path,
    title: r.meta.title || r.name
  }));
});
</script>
```

#### 3.2 返回顶部优化 🟢 **低优先级**

**优化方案**:

```vue
<!-- BackToTop.vue -->
<transition name="fade">
  <div 
    v-show="showButton"
    class="back-to-top"
    @click="scrollToTop"
  >
    <div class="progress-ring">
      <svg width="48" height="48">
        <circle
          :stroke-dasharray="circumference"
          :stroke-dashoffset="dashOffset"
          r="20"
          cx="24"
          cy="24"
        />
      </svg>
    </div>
    <i class="icon-arrow-up"></i>
  </div>
</transition>

<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue';

const showButton = ref(false);
const scrollProgress = ref(0);

// 滚动监听
const handleScroll = throttle(() => {
  const scrollTop = window.pageYOffset;
  const docHeight = document.documentElement.scrollHeight;
  const winHeight = window.innerHeight;
  
  showButton.value = scrollTop > 300;
  scrollProgress.value = scrollTop / (docHeight - winHeight);
}, 100);

// 进度环
const circumference = 2 * Math.PI * 20;
const dashOffset = computed(() => {
  return circumference * (1 - scrollProgress.value);
});

// 平滑滚动到顶部
const scrollToTop = () => {
  window.scrollTo({
    top: 0,
    behavior: 'smooth'
  });
};

onMounted(() => {
  window.addEventListener('scroll', handleScroll);
});

onUnmounted(() => {
  window.removeEventListener('scroll', handleScroll);
});
</script>
```

### 一、搜索体验优化

#### 1.1 智能搜索建议 🔴 **高优先级**

**现状问题**:
- 搜索建议不够智能
- 历史记录管理不便

**优化方案**:

```vue
<!-- SearchBar.vue -->
<div class="search-bar">
  <input
    v-model="keyword"
    type="text"
    placeholder="搜索小说、作者..."
    @input="handleInput"
    @focus="showSuggestions = true"
  />
  
  <!-- 搜索建议面板 -->
  <div v-show="showSuggestions" class="suggestions-panel">
    <!-- 热门搜索 -->
    <div v-if="!keyword" class="section">
      <div class="section-title">
        <i class="icon-fire"></i>热门搜索
      </div>
      <div class="tags">
        <span 
          v-for="item in hotSearches" 
          :key="item.id"
          class="tag"
          @click="search(item.keyword)"
        >
          {{ item.keyword }}
        </span>
      </div>
    </div>
    
    <!-- 搜索历史 -->
    <div v-if="!keyword && searchHistory.length" class="section">
      <div class="section-title">
        <i class="icon-clock"></i>搜索历史
        <button class="btn-clear" @click="clearHistory">清空</button>
      </div>
      <div class="history-list">
        <div 
          v-for="item in searchHistory" 
          :key="item.id"
          class="history-item"
          @click="search(item.keyword)"
        >
          <span class="keyword">{{ item.keyword }}</span>
          <button class="btn-delete" @click.stop="deleteHistory(item.id)">
            <i class="icon-close"></i>
          </button>
        </div>
      </div>
    </div>
    
    <!-- 实时建议 -->
    <div v-if="keyword && suggestions.length" class="section">
      <div class="section-title">
        <i class="icon-search"></i>搜索建议
      </div>
      <div class="suggestion-list">
        <div 
          v-for="item in suggestions" 
          :key="item.id"
          class="suggestion-item"
          @click="search(item.keyword)"
        >
          <!-- 高亮匹配文字 -->
          <span v-html="highlightKeyword(item.keyword, keyword)"></span>
          <span class="count">约{{ item.count }}本</span>
        </div>
      </div>
    </div>
    
    <!-- 搜索结果预览 -->
    <div v-if="keyword && novelPreviews.length" class="section">
      <div class="section-title">
        <i class="icon-book"></i>相关小说
      </div>
      <div class="novel-previews">
        <div 
          v-for="novel in novelPreviews" 
          :key="novel.id"
          class="novel-preview-item"
          @click="goToNovel(novel.id)"
        >
          <img :src="novel.cover" :alt="novel.title" class="cover-mini" />
          <div class="info">
            <div class="title">{{ novel.title }}</div>
            <div class="author">{{ novel.author }}</div>
          </div>
        </div>
      </div>
    </div>
  </div>
</div>

<script setup>
import { ref, watch } from 'vue';
import { debounce } from 'lodash-es';

const keyword = ref('');
const showSuggestions = ref(false);
const suggestions = ref([]);
const novelPreviews = ref([]);
const hotSearches = ref([]);
const searchHistory = ref([]);

// 实时搜索建议
const handleInput = debounce(async () => {
  if (!keyword.value) {
    suggestions.value = [];
    novelPreviews.value = [];
    return;
  }
  
  // 获取搜索建议
  suggestions.value = await fetchSuggestions(keyword.value);
  
  // 获取小说预览
  novelPreviews.value = await fetchNovelPreviews(keyword.value);
}, 300);

// 高亮关键词
const highlightKeyword = (text, keyword) => {
  const regex = new RegExp(`(${keyword})`, 'gi');
  return text.replace(regex, '<mark>$1</mark>');
};
</script>
```

**预期效果**:
- ✅ 搜索效率提升 60%
- ✅ 用户满意度提升 40%
- ✅ 搜索转化率提升 30%

#### 1.2 下拉刷新 🟡 **中优先级**

**优化方案**:

```vue
<!-- PullToRefresh.vue -->
<div 
  ref="container"
  class="pull-to-refresh-container"
  @touchstart="handleTouchStart"
  @touchmove="handleTouchMove"
  @touchend="handleTouchEnd"
>
  <!-- 刷新指示器 -->
  <div 
    class="pull-indicator"
    :style="{ transform: `translateY(${pullDistance}px)` }"
  >
    <div v-if="status === 'pulling'" class="indicator-content">
      <i class="icon-arrow-down"></i>
      <span>下拉刷新</span>
    </div>
    <div v-else-if="status === 'ready'" class="indicator-content">
      <i class="icon-arrow-up"></i>
      <span>释放刷新</span>
    </div>
    <div v-else-if="status === 'refreshing'" class="indicator-content">
      <i class="icon-loading spin"></i>
      <span>刷新中...</span>
    </div>
  </div>
  
  <!-- 内容 -->
  <slot></slot>
</div>

<script setup>
import { ref } from 'vue';

const container = ref(null);
const pullDistance = ref(0);
const status = ref('idle'); // idle, pulling, ready, refreshing

let startY = 0;
const threshold = 60; // 触发刷新的阈值

const handleTouchStart = (e) => {
  if (window.scrollY === 0) {
    startY = e.touches[0].clientY;
  }
};

const handleTouchMove = (e) => {
  if (startY === 0) return;
  
  const currentY = e.touches[0].clientY;
  const distance = currentY - startY;
  
  if (distance > 0) {
    // 阻止默认滚动
    e.preventDefault();
    
    // 计算拉动距离（添加阻力效果）
    pullDistance.value = Math.min(distance * 0.5, 100);
    
    // 更新状态
    if (pullDistance.value >= threshold) {
      status.value = 'ready';
    } else {
      status.value = 'pulling';
    }
  }
};

const handleTouchEnd = async () => {
  if (status.value === 'ready') {
    status.value = 'refreshing';
    
    try {
      await emit('refresh');
      showToast('刷新成功');
    } catch (error) {
      showToast('刷新失败，请重试');
    }
    
    status.value = 'idle';
  }
  
  // 重置
  pullDistance.value = 0;
  startY = 0;
  status.value = 'idle';
};
</script>
```
